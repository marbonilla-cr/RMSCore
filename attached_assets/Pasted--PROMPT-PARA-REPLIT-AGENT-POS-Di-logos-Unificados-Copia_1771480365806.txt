## LO QUE DEBES CONSTRUIR

### Componente 1: `<PayDialog />`
Un diálogo de pago con 3 paneles que se muestran SIMULTÁNEAMENTE en tablet/desktop
y como pasos deslizantes en mobile.

**Paneles:**
- Panel izquierdo (fijo): Resumen de la orden — lista de items, subtotal, impuestos, descuentos, total
- Panel central (fijo): Selección de método de pago (CASH / CARD / SINPE) + campos nombre y email del cliente
- Panel derecho (condicional): Se activa visualmente cuando el método es CASH — muestra grid de denominaciones, campo monto libre, cálculo de vuelto en tiempo real, botón COBRAR

**Regla crítica:** Ningún panel abre una ventana nueva ni navega a otra ruta.
Todo ocurre dentro del mismo diálogo. En mobile los paneles se deslizan con CSS transform.

---

### Componente 2: `<SplitDialog />`
Un diálogo de división de cuenta con 3 paneles SIMULTÁNEOS en tablet/desktop
y pasos deslizantes en mobile.

**Paneles:**
- Panel izquierdo: Lista de todos los items de la orden. Cada item tiene botón → para moverlo a la subcuenta activa. Los items movidos quedan visibles pero deshabilitados (opacidad reducida, borde punteado).
- Panel central: Subcuenta activa con tabs para cambiar entre subcuentas. Cada item asignado tiene botón ← para devolverlo a la cuenta principal. Botón "+ Nueva Subcuenta". Botón "SEPARAR TIQUETES".
- Panel derecho: Cards resumen de todas las subcuentas con sus totales. Botón "Pagar esta cuenta" por subcuenta. Botón "PAGAR TODO".

**Comportamiento del botón SEPARAR:**
1. Los paneles central y derecho vibran (animación CSS de 0.4s)
2. El diálogo hace flash verde (0.7s)
3. Se cierra SplitDialog
4. Se abre PayDialog automáticamente (setTimeout 320ms)

## ESTADO (hooks de React)

### usePayDialog.js
```javascript
export function usePayDialog() {
  const [isOpen, setIsOpen]     = useState(false);
  const [method, setMethod]     = useState(null);   // null | "CASH" | "CARD" | "SINPE"
  const [received, setReceived] = useState(0);
  const [mobileStep, setStep]   = useState(1);      // 1 | 2 | 3 (solo mobile)
  const [order, setOrder]       = useState(null);

  const total  = order?.total ?? 0;
  const change = received - total;
  const canPay = method === "CASH" ? change >= 0 : !!method;

  const open  = (orderData) => { setOrder(orderData); setIsOpen(true); reset(); };
  const close = () => setIsOpen(false);
  const reset = () => { setMethod(null); setReceived(0); setStep(1); };

  return { isOpen, method, received, mobileStep, total, change, canPay,
           open, close, setMethod, setReceived, setStep };
}
```

### useSplitDialog.js
```javascript
export function useSplitDialog() {
  const [isOpen, setIsOpen]       = useState(false);
  const [items, setItems]         = useState([]);
  const [subs, setSubs]           = useState([]);
  const [activeSub, setActiveSub] = useState(1);
  const [mobileStep, setStep]     = useState(1);

  const open = (orderItems) => {
    setItems(orderItems.map(i => ({ ...i, moved: false })));
    setSubs([{ id:1, name:"Sub #1", items:[] }, { id:2, name:"Sub #2", items:[] }]);
    setActiveSub(1); setStep(1); setIsOpen(true);
  };

  const moveToSub = (itemId) => {
    const item = items.find(i => i.id === itemId);
    if (!item || item.moved) return;
    setItems(p => p.map(i => i.id === itemId ? {...i, moved:true} : i));
    setSubs(p => p.map(s => s.id === activeSub
      ? { ...s, items: [...s.items, item] } : s));
  };

  const moveBack = (subId, itemId) => {
    setSubs(p => p.map(s => s.id === subId
      ? { ...s, items: s.items.filter(i => i.id !== itemId) } : s));
    setItems(p => p.map(i => i.id === itemId ? {...i, moved:false} : i));
  };

  const addSub = () => {
    const n = subs.length + 1;
    setSubs(p => [...p, { id:n, name:`Sub #${n}`, items:[] }]);
    setActiveSub(n);
  };

  return { isOpen, items, subs, activeSub, mobileStep,
           open, close: () => setIsOpen(false),
           moveToSub, moveBack, addSub, setActiveSub, setStep };
}
```

## LÓGICA DE SEPARACIÓN DE CUENTA

```javascript
// En SplitDialog.jsx
function handleSeparate() {
  const hasAnyItems = subs.some(s => s.items.length > 0);
  if (!hasAnyItems) return;

  // 1. Vibrar panels central y derecho
  const panelCentral = document.getElementById("split-panel-active");
  const panelDerecho = document.getElementById("split-panel-summary");

  [panelCentral, panelDerecho].forEach(el => {
    el.classList.remove("vibrating");
    void el.offsetWidth; // fuerza reflow para reiniciar animación
    el.classList.add("vibrating");
  });

  // 2. Después de 420ms → flash verde en el diálogo
  setTimeout(() => {
    [panelCentral, panelDerecho].forEach(el => el.classList.remove("vibrating"));
    document.getElementById("split-dialog").classList.add("flash-green");

    // 3. Después de 750ms → cerrar split, abrir pago
    setTimeout(() => {
      closeSplit();
      setTimeout(() => openPay(currentOrder), 320);
    }, 750);
  }, 420);
}
```
