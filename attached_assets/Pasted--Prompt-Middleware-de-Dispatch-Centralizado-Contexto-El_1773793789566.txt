## Prompt — Middleware de Dispatch Centralizado

**Contexto:** El check de `operationModeDispatch` está duplicado en 4 lugares inline. Hay que centralizarlo en un solo archivo como fuente de verdad.

---

**PASO 1 — Crear `server/middleware/dispatch.ts`:**

```typescript
/**
 * server/middleware/dispatch.ts
 * 
 * Fuente única de verdad para verificar si el modo despacho está activo.
 * Usar estas funciones en lugar de checks inline en routes.ts y qr-subaccount-routes.ts.
 */

import { getBusinessConfig } from "../storage";

/**
 * Helper async: retorna true si operationModeDispatch está activo para el tenant.
 */
export async function isDispatchEnabled(db: any): Promise<boolean> {
  try {
    const config = await getBusinessConfig(db);
    return config?.operationModeDispatch === true;
  } catch {
    return false;
  }
}

/**
 * Middleware Express: retorna 404 si despacho no está activo.
 * Usar en rutas públicas del QR de despacho.
 */
export function requireDispatchEnabled(
  req: any,
  res: any,
  next: any
): void {
  isDispatchEnabled(req.db)
    .then(enabled => {
      if (!enabled) {
        return res.status(404).json({
          message: "El servicio de despacho no está disponible en este restaurante."
        });
      }
      next();
    })
    .catch(() => {
      res.status(500).json({ message: "Error verificando configuración de despacho" });
    });
}
```
