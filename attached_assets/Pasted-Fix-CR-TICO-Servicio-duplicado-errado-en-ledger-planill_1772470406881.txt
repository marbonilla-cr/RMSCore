T002: Limpiar datos históricos — backup, dedupe, recalcular, snapshot
BLOCKED BY: T001
FILES: SQL directo en BD (one-time)

Paso 1 — Backup:
CREATE TABLE service_charge_ledger_backup_20260302 AS
SELECT * FROM service_charge_ledger;

Paso 2 — Eliminar duplicados PAID (window function segura):
WITH ranked AS (
  SELECT id,
         ROW_NUMBER() OVER (PARTITION BY order_id, order_item_id ORDER BY id DESC) AS rn
  FROM service_charge_ledger
  WHERE status = 'PAID'
)
DELETE FROM service_charge_ledger
WHERE id IN (SELECT id FROM ranked WHERE rn > 1);

Paso 3 — Recalcular service_amount con fórmula correcta:
UPDATE service_charge_ledger
SET service_amount = ROUND(
  (base_amount_snapshot::numeric * rate_snapshot::numeric) / (1 + rate_snapshot::numeric),
  2
)
WHERE status = 'PAID';

Paso 4 — Set includes_service_snapshot derivado (no mentir):
UPDATE service_charge_ledger
SET includes_service_snapshot = (service_amount::numeric > 0)
WHERE status = 'PAID';
