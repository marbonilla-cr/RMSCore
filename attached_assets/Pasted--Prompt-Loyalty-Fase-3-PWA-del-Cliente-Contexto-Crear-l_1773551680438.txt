## Prompt — Loyalty Fase 3: PWA del Cliente

**Contexto:** Crear la Progressive Web App del cliente en `loyalty.rmscore.app`. Es una app separada dentro del mismo servidor Express — cuando el host es `loyalty.rmscore.app`, el servidor sirve el frontend de loyalty en lugar del RMS operativo. El cliente se registra con Google, ve sus puntos y su historial.

---

### PARTE A — Detección de host en el servidor

En `server/index.ts` o `server/routes.ts`, agregar lógica para servir el frontend de loyalty cuando el host sea `loyalty.rmscore.app`:

```typescript
// Antes de las rutas del RMS, detectar si es loyalty
app.use((req, res, next) => {
  const host = req.headers.host || "";
  if (host.startsWith("loyalty.")) {
    req.isLoyaltyApp = true;
  }
  next();
});
```

Agregar en los tipos de Express:
```typescript
declare global {
  namespace Express {
    interface Request {
      isLoyaltyApp?: boolean;
    }
  }
}
```

Servir el HTML de loyalty para cualquier ruta GET en el host loyalty:
```typescript
app.get("*", (req, res, next) => {
  if (req.isLoyaltyApp && !req.path.startsWith("/api/")) {
    return res.sendFile(path.join(__dirname, "../client-loyalty/index.html"));
  }
  next();
});
```

const API_BASE = "";
const GOOGLE_CLIENT_ID = "580542734412-01e4igcmvupjmihjm2tq9fk114r8m4oq.apps.googleusercontent.com";

// Estado de la app
let state = {
  screen: "loading", // loading | login | home | restaurant | history
  customer: null,
  token: null,
  loyaltyAccounts: [],
  currentTenantId: null,
  currentAccount: null,
  events: [],
};

// Persistencia local
function saveSession() {
  if (state.customer && state.token) {
    localStorage.setItem("loyalty_session", JSON.stringify({
      customer: state.customer,
      token: state.token,
    }));
  }
}

function loadSession() {
  try {
    const saved = localStorage.getItem("loyalty_session");
    if (saved) {
      const { customer, token } = JSON.parse(saved);
      state.customer = customer;
      state.token = token;
      return true;
    }
  } catch (e) {}
  return false;
}

function clearSession() {
  localStorage.removeItem("loyalty_session");
  state.customer = null;
  state.token = null;
}

// API calls
async function apiGet(path) {
  const res = await fetch(`${API_BASE}/api/loyalty${path}`, {
    headers: { "X-Loyalty-Token": state.token || "" }
  });
  if (!res.ok) throw new Error(await res.text());
  return res.json();
}

async function apiPost(path, body) {
  const res = await fetch(`${API_BASE}/api/loyalty${path}`, {
    method: "POST",
    headers: { 
      "Content-Type": "application/json",
      "X-Loyalty-Token": state.token || ""
    },
    body: JSON.stringify(body),
  });
  if (!res.ok) throw new Error(await res.text());
  return res.json();
}

// Google Sign-In
function initGoogleSignIn() {
  google.accounts.id.initialize({
    client_id: GOOGLE_CLIENT_ID,
    callback: handleGoogleResponse,
    auto_select: false,
  });
}

async function handleGoogleResponse(response) {
  try {
    showLoading("Iniciando sesión...");
    const result = await apiPost("/auth/google", { idToken: response.credential });
    state.customer = result.customer;
    state.token = result.token;
    saveSession();
    await loadHomeData();
    renderHome();
  } catch (err) {
    showError("Error al iniciar sesión: " + err.message);
    renderLogin();
  }
}

async function loadHomeData() {
  try {
    // Cargar cuentas de loyalty del cliente en todos los tenants
    const accounts = await apiGet(`/customers/${state.customer.id}/accounts`);
    state.loyaltyAccounts = accounts;
  } catch (e) {
    state.loyaltyAccounts = [];
  }
}

async function loadRestaurantDetail(tenantId) {
  showLoading("Cargando historial...");
  try {
    // Necesita header X-Tenant-Id para que el backend sepa qué tenant
    const res = await fetch(`/api/loyalty/customers/${state.customer.id}`, {
      headers: { 
        "X-Loyalty-Token": state.token,
        "X-Tenant-Id": String(tenantId)
      }
    });
    const data = await res.json();
    renderRestaurantDetail(data, tenantId);
  } catch (err) {
    showError("Error cargando historial");
  }
}

### PARTE C — Endpoint adicional en backend

En `server/loyalty-routes.ts`, agregar:

```typescript
// GET /api/loyalty/customers/:id/accounts — cuentas en todos los tenants
app.get("/api/loyalty/customers/:id/accounts", async (req: any, res: any) => {
  try {
    const customerId = parseInt(req.params.id);
    const result = await db.execute(
      `SELECT la.*, t.business_name, 
              MAX(le.created_at) as last_visit
       FROM public.loyalty_accounts la
       JOIN public.tenants t ON t.id = la.tenant_id
       LEFT JOIN public.loyalty_events le ON le.customer_id = la.customer_id 
         AND le.tenant_id = la.tenant_id
       WHERE la.customer_id = $1
       GROUP BY la.id, t.business_name
       ORDER BY last_visit DESC NULLS LAST`,
      [customerId]
    );
    res.json(result.rows);
  } catch (err: any) {
    res.status(500).json({ message: err.message });
  }
});
```

### PARTE D — Servir archivos estáticos de loyalty

En `server/index.ts`, agregar antes de las rutas:

```typescript
import path from "path";

// Servir archivos de loyalty cuando el host sea loyalty.*
app.use((req, res, next) => {
  const host = req.headers.host || "";
  if (host.startsWith("loyalty.") && !req.path.startsWith("/api/")) {
    const loyaltyDir = path.join(process.cwd(), "client-loyalty");
    if (req.path === "/" || !req.path.includes(".")) {
      return res.sendFile(path.join(loyaltyDir, "index.html"));
    }
    return res.sendFile(path.join(loyaltyDir, req.path), (err) => {
      if (err) res.sendFile(path.join(loyaltyDir, "index.html"));
    });
  }
  next();
});
```
