### PARTE C — Job de auto-cancelación

En `server/hr-jobs.ts` o crear `server/dispatch-jobs.ts`, agregar job que corre cada minuto:

```typescript
export async function startDispatchBackgroundJobs() {
  setInterval(async () => {
    try {
      const { Pool } = require("pg");
      const publicPool = new Pool({ connectionString: process.env.DATABASE_URL, max: 1 });
      const { rows: tenants } = await publicPool.query(
        `SELECT schema_name FROM public.tenants WHERE is_active = true`
      );
      await publicPool.end();

      for (const { schema_name } of tenants) {
        try {
          const tdb = getTenantDb(schema_name);
          const config = await getBusinessConfig(tdb);
          if (!config?.operationModeDispatch) continue;

          const timeoutMinutes = config.dispatchOrderTimeoutMinutes || 15;
          const cutoff = new Date(Date.now() - timeoutMinutes * 60 * 1000);

          // Auto-cancelar órdenes PENDING_PAYMENT vencidas
          const cancelled = await tdb.update(orders)
            .set({ 
              dispatchStatus: "CANCELLED",
              status: "CANCELLED",
            })
            .where(
              and(
                eq(orders.orderMode, "DISPATCH"),
                eq(orders.dispatchStatus, "PENDING_PAYMENT"),
                lt(orders.createdAt, cutoff)
              )
            )
            .returning({ id: orders.id, transactionCode: orders.transactionCode });

          for (const order of cancelled) {
            console.log(`[Dispatch] Auto-cancelada orden ${order.transactionCode} en ${schema_name}`);
            // Emitir WebSocket para notificar al cliente
            broadcast({ type: "dispatch_order_cancelled", orderId: order.id, transactionCode: order.transactionCode });
          }
        } catch (err: any) {
          console.error(`[Dispatch] Error en schema ${schema_name}:`, err.message);
        }
      }
    } catch (err: any) {
      console.error("[Dispatch] Job error:", err.message);
    }
  }, 60 * 1000); // cada minuto
  console.log("[Dispatch] Background jobs started (auto-cancel every 1min)");
}
```

Llamar `startDispatchBackgroundJobs()` desde `server/index.ts`.

```typescript
app.post("/api/pos/orders/:id/cancel-dispatch", requireAuth, async (req, res) => {
  try {
    const [order] = await req.db.update(orders)
      .set({ dispatchStatus: "CANCELLED", status: "CANCELLED" })
      .where(
        and(
          eq(orders.id, parseInt(req.params.id)),
          eq(orders.dispatchStatus, "PENDING_PAYMENT")
        )
      )
      .returning();
    if (!order) return res.status(404).json({ message: "Orden no encontrada o ya procesada" });
    broadcast({ type: "dispatch_order_cancelled", orderId: order.id, transactionCode: order.transactionCode });
    res.json(order);
  } catch (err: any) {
    res.status(500).json({ message: err.message });
  }
});
```
