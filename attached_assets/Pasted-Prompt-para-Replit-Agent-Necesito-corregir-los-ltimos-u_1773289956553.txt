CAMBIO 4 — retryPendingSync() en server/quickbooks.ts
Extraer la lógica actual a una función interna retryPendingSyncForTenant y hacer que retryPendingSync itere los tenants activos:

// Nueva función interna — misma lógica de hoy, recibe dbInstance:
async function retryPendingSyncForTenant(tdb: typeof db): Promise<number> {
  const config = await getQboConfig(tdb);
  if (!config || !config.isConnected) return 0;

  await tdb.update(qboSyncLog).set({
    status: "PENDING",
    attempts: 0,
    errorMessage: null,
    nextRetryAt: null,
  }).where(
    sql`${qboSyncLog.status} = 'FAILED' AND ${qboSyncLog.errorMessage} LIKE '%documento duplicado%'`
  );

  const now = new Date();
  const pendingLogs = await tdb.select().from(qboSyncLog)
    .where(
      sql`(${qboSyncLog.status} = 'PENDING') OR
          (${qboSyncLog.status} = 'FAILED' AND ${qboSyncLog.nextRetryAt} <= ${now}
           AND ${qboSyncLog.attempts} < 5)`
    )
    .limit(20);

  const backoffMinutes = [5, 15, 60, 240, 1440];
  let processed = 0;
  for (const log of pendingLogs) {
    try {
      await createSalesReceipt(log.orderId, log.paymentId, tdb);
      processed++;
    } catch (err: any) {
      const newAttempts = (log.attempts || 0) + 1;
      if (newAttempts >= 5) {
        await tdb.update(qboSyncLog).set({ status: "ABANDONED", attempts: newAttempts, errorMessage: err.message }).where(eq(qboSyncLog.id, log.id));
      } else {
        const nextRetry = new Date(Date.now() + backoffMinutes[newAttempts - 1] * 60 * 1000);
        await tdb.update(qboSyncLog).set({ status: "FAILED", attempts: newAttempts, errorMessage: err.message, nextRetryAt: nextRetry }).where(eq(qboSyncLog.id, log.id));
      }
    }
  }
  return processed;
}

// Función pública — ahora itera tenants:
export async function retryPendingSync(): Promise<number> {
  const { Pool } = await import("pg");
  const publicPool = new Pool({ connectionString: process.env.DATABASE_URL, max: 1 });
  let totalProcessed = 0;
  try {
    const { rows: tenants } = await publicPool.query(
      `SELECT schema_name FROM public.tenants WHERE is_active = true`
    );
    const schemas = tenants.map((t: any) => t.schema_name);
    if (schemas.length === 0) schemas.push(process.env.TENANT_SCHEMA || "public");
    for (const schema of schemas) {
      try {
        const { getTenantDb } = await import("./db-tenant");
        const tdb = getTenantDb(schema);
        totalProcessed += await retryPendingSyncForTenant(tdb);
      } catch (err: any) {
        console.error(`[QBO] Retry error for schema ${schema}:`, err.message);
      }
    }
  } finally {
    await publicPool.end();
  }
  return totalProcessed;
}
