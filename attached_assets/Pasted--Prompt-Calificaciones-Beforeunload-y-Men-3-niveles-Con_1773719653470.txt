### PARTE B — Backend

**1. Endpoint para guardar review:**

```typescript
app.post("/api/qr/:tableCode/review", async (req, res) => {
  try {
    const { orderId, rating, comment, customerId } = req.body;
    if (!orderId || !rating) return res.status(400).json({ message: "orderId y rating requeridos" });

    const config = await getBusinessConfig(req.db);
    const businessDate = await getBusinessDate(req.tenantSchema);

    // Verificar que la orden existe y no tiene review
    const existing = await req.db.select().from(orderReviews)
      .where(eq(orderReviews.orderId, orderId));
    if (existing.length > 0) return res.status(409).json({ message: "Ya existe una calificación para esta orden" });

    const pointsAwarded = config?.reviewPoints || 50;

    // Guardar review
    const [review] = await req.db.insert(orderReviews).values({
      orderId,
      customerId: customerId || null,
      rating,
      comment: comment || null,
      pointsAwarded: String(pointsAwarded),
      businessDate,
    }).returning();

    // Acreditar puntos si hay cliente loyalty
    if (customerId) {
      const { db: publicDb } = await import("./db");
      const existing = await publicDb.select().from(loyaltyAccounts)
        .where(and(
          eq(loyaltyAccounts.customerId, customerId),
          eq(loyaltyAccounts.tenantId, req.tenantId)
        ));
      if (existing.length > 0) {
        await publicDb.update(loyaltyAccounts).set({
          pointsBalance: String(parseFloat(existing[0].pointsBalance) + pointsAwarded),
          lifetimePoints: String(parseFloat(existing[0].lifetimePoints) + pointsAwarded),
          updatedAt: new Date(),
        }).where(eq(loyaltyAccounts.id, existing[0].id));

        await publicDb.insert(loyaltyEvents).values({
          customerId,
          tenantId: req.tenantId,
          eventType: "REVIEW",
          points: String(pointsAwarded),
          orderId,
          description: `+${pointsAwarded} puntos por calificación ⭐`.repeat(rating).slice(0, 50),
        });
      }
    }

    // Si rating <= 3, enviar email a gerencia
    if (rating <= 3 && config?.reviewEmail) {
      const { sendEmail } = await import("./services/email-service");
      await sendEmail({
        to: config.reviewEmail,
        subject: `⚠️ Calificación baja: ${rating}/5 estrellas`,
        html: `
          <h2>Calificación recibida: ${"⭐".repeat(rating)}</h2>
          <p><strong>Orden:</strong> #${orderId}</p>
          <p><strong>Comentario:</strong> ${comment || "Sin comentario"}</p>
          <p><strong>Fecha:</strong> ${businessDate}</p>
        `,
      });
    }

    res.json({ review, pointsAwarded });
  } catch (err: any) {
    res.status(500).json({ message: err.message });
  }
});
```

**2. WebSocket: evento `review_prompt` a los 30 minutos para QR de mesa:**

En el handler de `kitchen_item_status_changed` o donde se detecta que todos los items de una orden están READY, agregar:

```typescript
// Si la orden es QR (no despacho), programar review prompt a los 30 min

  setTimeout(() => {
    broadcast({
      type: "review_prompt",
      orderId: order.id,
      tableCode: order.tableCode,
      pointsAwarded: config.reviewPoints,
    });

```

**3. Beforeunload — evitar cierre accidental:**

Agregar en el `useEffect` principal del componente:

```typescript
useEffect(() => {
  const handleBeforeUnload = (e: BeforeUnloadEvent) => {
    if (["menu", "review", "sent", "dispatch"].includes(screen)) {
      e.preventDefault();
      e.returnValue = "";
    }
  };
  window.addEventListener("beforeunload", handleBeforeUnload);
  return () => window.removeEventListener("beforeunload", handleBeforeUnload);
}, [screen]);
```
